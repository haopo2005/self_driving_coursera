#### 4. Measurement Update #####################################################################

################################################################################################
# Since we'll need a measurement update for both the GNSS and the LIDAR data, let's make
# a function for it.
################################################################################################
def measurement_update(sensor_var, p_cov_check, y_k, p_check, v_check, q_check):
    # 3.1 Compute Kalman Gain
    K_k = p_cov_check @ h_jac.T @ np.linalg.inv(h_jac @ p_cov_check @ h_jac.T + np.eye(3)*sensor_var)
    # 3.2 Compute error state
    delta_x_k = K_k @ (y_k - p_check).T
    # 3.3 Correct predicted state
    p_hat = p_check + np.ravel(delta_x_k[0:3])
    v_hat = v_check + np.ravel(delta_x_k[3:6])
    q_Phi = Quaternion(axis_angle = angle_normalize(np.ravel(delta_x_k[6:9])))
    q_hat = q_Phi.quat_mult_right(q_check)
    
    # 3.4 Compute corrected covariance
    p_cov_hat = (np.eye(9)-K_k@h_jac)@p_cov_check
    
    return p_hat, v_hat, q_hat, p_cov_hat

#### 5. Main Filter Loop #######################################################################

################################################################################################
# Now that everything is set up, we can start taking in the sensor data and creating estimates
# for our state in a loop.
################################################################################################
Q = np.zeros((6,6))
Q[0,0] = var_imu_f
Q[1,1] = var_imu_f
Q[2,2] = var_imu_f
Q[3,3] = var_imu_w
Q[4,4] = var_imu_w
Q[5,5] = var_imu_w

for k in range(1, imu_f.data.shape[0]):  # start at 1 b/c we have initial prediction from gt
    delta_t = imu_f.t[k] - imu_f.t[k - 1]
    
    # 1. Update state with IMU inputs

    # 1.1 Linearize the motion model and compute Jacobians
    Cns  = Quaternion(*q_est[k-1])
    CnsR = Cns.to_mat()
    p_check = p_est[k-1,:] + delta_t * v_est[k-1,:] + delta_t**2/2*(CnsR@imu_f.data[k-1] + g)
    v_check = v_est[k-1,:] + delta_t * (CnsR@imu_f.data[k-1] + g)
    q_check = Quaternion(axis_angle = angle_normalize(delta_t*imu_w.data[k-1])).quat_mult_right(q_est[k-1])
    # 2. Propagate uncertainty
    F_jac = np.eye(9)
    F_jac[0:3,3:6] = np.eye(3)*delta_t
    F_jac[3:6,6:9] = -CnsR@skew_symmetric(imu_f.data[k-1])*delta_t
    
    p_cov_check = F_jac @ p_cov[k-1] @ F_jac.T + l_jac @ (Q*(delta_t**2)) @ l_jac.T
    # 3. Check availability of GNSS and LIDAR measurements
    idx_gnss  = np.logical_and(gnss.t>=imu_f.t[k - 1], gnss.t<imu_f.t[k])
    idx_lidar = np.logical_and(lidar.t>=imu_f.t[k - 1], lidar.t<imu_f.t[k])
    tmp_gnss  = gnss.data[idx_gnss]
    tmp_lidar = lidar.data[idx_lidar]
    
    if len(tmp_gnss) != 0:
        sensor_var = var_gnss
        y_k = tmp_gnss
    elif len(tmp_lidar) != 0:
        sensor_var = var_lidar
        y_k = tmp_lidar
    else:
        p_est[k,:], v_est[k,:], q_est[k,:], p_cov[k,:,:] = p_check, v_check, q_check, p_cov_check
        continue
    
    
    # Update states (save)
    p_est[k,:], v_est[k,:], q_est[k,:], p_cov[k,:,:] = measurement_update(sensor_var, p_cov_check, y_k, p_check, v_check, q_check)
